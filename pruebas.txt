#Se crea Disco 1,2,3 y 4
mkdisk -size=25 -u=m -path=/home/cristian/PruebAsMIA/Disco1.dk
mkdisk -size=2000 -u=k -path=/home/cristian/PruebAsMIA/Disco2.dk
mkdisk -size=10 -path=/home/cristian/PruebAsMIA/Disco3.dk

#Se elimina disco 3
rmdisk -path=/home/cristian/PruebAsMIA/Disco3.dk

#Crea una partición primaria llamada Particion1 en Disco1.disk
fdisk -Size=3000 -path=/home/cristian/PruebAsMIA/Disco1.dk -name=Particion1
mount -name=Particion1 -path=/home/cristian/PruebAsMIA/Disco1.dk

#Crea una partición primaria llamada Particion2 en Disco1.disk
fdisk -Size=5 -u=m -path=/home/cristian/PruebAsMIA/Disco1.dk -name=Particion2

#Crea una partición extendida llamada Particion4 en Disco1.disk
fdisk -Size=10 -u=m  -type=E -path=/home/cristian/PruebAsMIA/Disco1.dk -name=Particion4

#Crea una partición logica llamada ParticionL1 en Particion4 en Disco1.disk
fdisk -Size=300 -type=L -path=/home/cristian/PruebAsMIA/Disco1.dk -name=ParticionL1
#Crea una partición logica llamada ParticionL2 en Particion4 en Disco1.disk
fdisk -Size=300 -type=L -path=/home/cristian/PruebAsMIA/Disco1.dk -name=ParticionL2

pause

fdisk -Size=300 -type=L -path=/home/cristian/PruebAsMIA/Disco1.dk -name=ParticionL3
fdisk -delete=full -path=/home/cristian/PruebAsMIA/Disco1.dk -name=ParticionL3
